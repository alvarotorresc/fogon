# Fogon MVP Implementation Plan

> **For Claude:** REQUIRED SUB-SKILL: Use superpowers:executing-plans to implement this plan task-by-task.

**Goal:** Build Fogon MVP — app de cocina colaborativa con lista de la compra, despensa, planificador semanal y recetario, todos módulos manuales e independientes (v0.1). Sin integraciones entre módulos.

**Architecture:** Monorepo pnpm workspaces. Mobile (Expo) accede a Supabase directamente para auth/CRUD/realtime (RLS policies). NestJS scaffolded pero solo usado para generación de invite codes. Cada módulo funciona de forma autónoma — sin conexiones cross-módulo en v0.1.

**Tech Stack:** Expo SDK 53, expo-router v4, NativeWind v4, Zustand, @tanstack/react-query v5, @supabase/supabase-js v2, i18next + react-i18next, NestJS 11, Astro 5, pnpm workspaces, TypeScript strict

**Step 2: Crear RLS migration** (`20260302000002_fogon_rls.sql`)

```sql
-- Helper: es miembro del hogar
create or replace function public.is_household_member(p_household_id uuid)
returns boolean as $$
  select exists (
    select 1 from public.household_members
    where household_id = p_household_id and user_id = auth.uid()
  )
$$ language sql security definer stable;

-- Households RLS
alter table public.households enable row level security;
create policy "household_select" on public.households for select using (is_household_member(id));
create policy "household_insert" on public.households for insert with check (created_by = auth.uid());
create policy "household_update" on public.households for update using (created_by = auth.uid());

-- Members RLS
alter table public.household_members enable row level security;
create policy "members_select" on public.household_members for select using (is_household_member(household_id));
create policy "members_insert" on public.household_members for insert with check (user_id = auth.uid());
create policy "members_delete" on public.household_members for delete using (user_id = auth.uid());

-- Shopping items RLS
alter table public.shopping_items enable row level security;
create policy "shopping_select" on public.shopping_items for select using (is_household_member(household_id));
create policy "shopping_insert" on public.shopping_items for insert with check (is_household_member(household_id) and added_by = auth.uid());
create policy "shopping_update" on public.shopping_items for update using (is_household_member(household_id));
create policy "shopping_delete" on public.shopping_items for delete using (is_household_member(household_id));

-- Pantry items RLS
alter table public.pantry_items enable row level security;
create policy "pantry_select" on public.pantry_items for select using (is_household_member(household_id));
create policy "pantry_insert" on public.pantry_items for insert with check (is_household_member(household_id) and added_by = auth.uid());
create policy "pantry_update" on public.pantry_items for update using (is_household_member(household_id));
create policy "pantry_delete" on public.pantry_items for delete using (is_household_member(household_id));

-- Recipes RLS
alter table public.recipes enable row level security;
-- Curated (household_id IS NULL): todos pueden leer
create policy "recipes_select_curated" on public.recipes for select using (household_id is null);
-- Propias: solo miembros del hogar o is_public
create policy "recipes_select_own" on public.recipes for select using (is_household_member(household_id));
create policy "recipes_select_public" on public.recipes for select using (is_public = true);
create policy "recipes_insert" on public.recipes for insert with check (is_household_member(household_id) and created_by = auth.uid());
create policy "recipes_update" on public.recipes for update using (is_household_member(household_id));
create policy "recipes_delete" on public.recipes for delete using (is_household_member(household_id));

-- Recipe ingredients/steps RLS (inherit via recipe)
alter table public.recipe_ingredients enable row level security;
create policy "ingredients_select" on public.recipe_ingredients for select
  using (exists (select 1 from public.recipes r where r.id = recipe_id and (r.household_id is null or is_household_member(r.household_id) or r.is_public)));
create policy "ingredients_insert" on public.recipe_ingredients for insert
  with check (exists (select 1 from public.recipes r where r.id = recipe_id and is_household_member(r.household_id)));
create policy "ingredients_update" on public.recipe_ingredients for update
  using (exists (select 1 from public.recipes r where r.id = recipe_id and is_household_member(r.household_id)));
create policy "ingredients_delete" on public.recipe_ingredients for delete
  using (exists (select 1 from public.recipes r where r.id = recipe_id and is_household_member(r.household_id)));

alter table public.recipe_steps enable row level security;
create policy "steps_select" on public.recipe_steps for select
  using (exists (select 1 from public.recipes r where r.id = recipe_id and (r.household_id is null or is_household_member(r.household_id) or r.is_public)));
create policy "steps_insert" on public.recipe_steps for insert
  with check (exists (select 1 from public.recipes r where r.id = recipe_id and is_household_member(r.household_id)));
create policy "steps_update" on public.recipe_steps for update
  using (exists (select 1 from public.recipes r where r.id = recipe_id and is_household_member(r.household_id)));
create policy "steps_delete" on public.recipe_steps for delete
  using (exists (select 1 from public.recipes r where r.id = recipe_id and is_household_member(r.household_id)));

-- Meal plan RLS
alter table public.meal_plan_entries enable row level security;
create policy "mealplan_select" on public.meal_plan_entries for select using (is_household_member(household_id));
create policy "mealplan_insert" on public.meal_plan_entries for insert with check (is_household_member(household_id) and created_by = auth.uid());
create policy "mealplan_update" on public.meal_plan_entries for update using (is_household_member(household_id));
create policy "mealplan_delete" on public.meal_plan_entries for delete using (is_household_member(household_id));
```

```typescript
import { useQuery, useMutation, useQueryClient } from '@tanstack/react-query';
import { supabase } from '@/lib/supabase';
import { useHouseholdStore } from '@/store/householdStore';
import type { ShoppingItem } from '@fogon/types';

const QUERY_KEY = 'shopping_items';

export function useShoppingList() {
  const { household } = useHouseholdStore();

  return useQuery({
    queryKey: [QUERY_KEY, household?.id],
    queryFn: async (): Promise<ShoppingItem[]> => {
      if (!household) return [];
      const { data, error } = await supabase
        .from('shopping_items')
        .select('*, members:household_members(display_name, user_id)')
        .eq('household_id', household.id)
        .order('created_at', { ascending: true });
      if (error) throw error;
      return (data ?? []).map((row) => ({
        id: row.id,
        name: row.name,
        quantity: row.quantity,
        category: row.category,
        isDone: row.is_done,
        doneByName: row.done_by ? row.members?.find((m: any) => m.user_id === row.done_by)?.display_name ?? null : null,
        addedByName: row.members?.find((m: any) => m.user_id === row.added_by)?.display_name ?? 'Alguien',
        createdAt: row.created_at,
      }));
    },
    enabled: !!household,
  });
}
```

```typescript
import { useEffect, useRef } from 'react';
import { useQueryClient } from '@tanstack/react-query';
import { supabase } from '@/lib/supabase';
import { useHouseholdStore } from '@/store/householdStore';

export function useRealtimeSync() {
  const { household } = useHouseholdStore();
  const qc = useQueryClient();
  const channelRef = useRef<ReturnType<typeof supabase.channel> | null>(null);

  useEffect(() => {
    if (!household) return;

    channelRef.current = supabase
      .channel(`shopping:${household.id}`)
      .on('postgres_changes', { event: '*', schema: 'public', table: 'shopping_items', filter: `household_id=eq.${household.id}` },
        () => { qc.invalidateQueries({ queryKey: ['shopping_items', household.id] }); }
      )
      .subscribe();

    return () => { channelRef.current?.unsubscribe(); };
  }, [household?.id]);
}
```

```typescript
export function useRecipes() {

      const { data, error } = await supabase
        .from('recipes')
        .select(`*, recipe_ingredients(*), recipe_steps(*)`)
        .or(`household_id.is.null,household_id.eq.${household?.id}`)
        .order('created_at', { ascending: false });
      if (error) throw error;
      return (data ?? []).map((row) => ({

        prepTimeMinutes: row.prep_time_minutes, imageUrl: row.image_url,
        isPublic: row.is_public, isCurated: row.household_id === null,
        ingredients: (row.recipe_ingredients ?? []).sort((a: any, b: any) => a.position - b.position).map((i: any) => ({
          id: i.id, name: i.name, quantity: i.quantity, unit: i.unit,
        })),
        steps: (row.recipe_steps ?? []).sort((a: any, b: any) => a.step_number - b.step_number).map((s: any) => ({
          id: s.id, stepNumber: s.step_number, description: s.description,
        })),
      }));
    },
    enabled: !!household,
  });

export function useCreateRecipe() {
  const qc = useQueryClient();
  const { household } = useHouseholdStore();

  return useMutation({
    mutationFn: async (data: {
      title: string;
      description?: string;
      prepTimeMinutes?: number;
      isPublic?: boolean;
      ingredients: Array<{ name: string; quantity?: string; unit?: string }>;
      steps: Array<{ description: string }>;
    }) => {
      const { data: { user } } = await supabase.auth.getUser();
      const { data: recipe, error } = await supabase.from('recipes').insert({
        household_id: household!.id, title: data.title, description: data.description ?? null,
        prep_time_minutes: data.prepTimeMinutes ?? null, is_public: data.isPublic ?? false, created_by: user!.id,
      }).select().single();
      if (error) throw error;

      // Insertar ingredientes y pasos
      if (data.ingredients.length > 0) {
        const { error: ingError } = await supabase.from('recipe_ingredients').insert(
          data.ingredients.map((ing, i) => ({ recipe_id: recipe.id, name: ing.name, quantity: ing.quantity ?? null, unit: ing.unit ?? null, position: i }))
        );
        if (ingError) throw ingError;
      }
      if (data.steps.length > 0) {
        const { error: stepError } = await supabase.from('recipe_steps').insert(
          data.steps.map((step, i) => ({ recipe_id: recipe.id, step_number: i + 1, description: step.description }))
        );
        if (stepError) throw stepError;
      }
    },
    onSuccess: () => qc.invalidateQueries({ queryKey: ['recipes'] }),
  });
```

**Step 2: Crear household.service.ts** con lógica de validación de invite code

```typescript
import { Injectable, NotFoundException, ConflictException } from '@nestjs/common';
import { createClient } from '@supabase/supabase-js';
import { AVATAR_COLORS } from './constants';

@Injectable()
export class HouseholdService {
  private readonly supabase = createClient(
    process.env.SUPABASE_URL!,
    process.env.SUPABASE_SERVICE_ROLE_KEY!, // Service role para bypassar RLS
  );

  async joinByInviteCode(userId: string, inviteCode: string, displayName: string): Promise<void> {
    // 1. Buscar household por invite_code
    const { data: household, error: findError } = await this.supabase
      .from('households')
      .select('id')
      .eq('invite_code', inviteCode.toUpperCase())
      .single();

    if (findError || !household) throw new NotFoundException('Código de invitación no válido');

    // 2. Comprobar si ya es miembro
    const { data: existing } = await this.supabase
      .from('household_members')
      .select('id')
      .eq('household_id', household.id)
      .eq('user_id', userId)
      .single();

    if (existing) throw new ConflictException('Ya eres miembro de este hogar');

    // 3. Asignar color de avatar (pick del array según count)
    const { count } = await this.supabase
      .from('household_members')
      .select('*', { count: 'exact', head: true })
      .eq('household_id', household.id);

    const avatarColor = AVATAR_COLORS[(count ?? 0) % AVATAR_COLORS.length];

    // 4. Insertar miembro
    const { error: insertError } = await this.supabase.from('household_members').insert({
      household_id: household.id, user_id: userId,
      display_name: displayName, avatar_color: avatarColor, role: 'member',
    });

    if (insertError) throw new Error(insertError.message);
  }
}
```

## Notas de implementación

### Bug conocido: NativeWind + Pressable
NativeWind con `jsxImportSource: 'nativewind'` rompe function styles en `Pressable`. Usar children render prop:
```tsx
<Pressable ...>
  {({ pressed }) => <Text style={{ opacity: pressed ? 0.7 : 1 }}>...</Text>}
</Pressable>
```

### Bottom sheets sin librería
Para MVP, usar `Modal` nativa con animación slide-up:
```tsx
<Modal transparent animationType="slide" visible={visible} onRequestClose={onClose}>
  <Pressable style={{ flex: 1 }} onPress={onClose} />
  <View className="bg-bg-elevated rounded-t-2xl p-5">
    {children}
  </View>
</Modal>
```

### Variables de entorno
- Mobile: `EXPO_PUBLIC_SUPABASE_URL` + `EXPO_PUBLIC_SUPABASE_ANON_KEY` (en `.env.local`)

### Recetas curadas
